# 微信定时消息发送工具 - 需求文档

## 一、功能需求

使用Python实现定时发送微信消息的工具，支持精确时间控制（毫秒级）和多任务管理。

---

## 二、技术方案

### 方案选择：PyAutoGUI模拟键盘操作

**核心原理**：
- 使用PyAutoGUI模拟键盘输入和鼠标操作
- 通过NTP协议校准本地时间，确保发送精度
- 提前准备消息内容，在精确时间点执行发送动作

**为什么选这个方案**：
- ✅ 无需微信API，个人微信可用
- ✅ 实现简单，稳定性好
- ✅ 模拟真实用户操作，不易被检测
- ❌ 需要保持微信PC版运行
- ❌ 不能后台运行（窗口可以被最小化，但不能关闭）

### 4.1 时间校准模块

```python
import ntplib
from datetime import datetime, timezone

class TimeCalibrator:
    """时间校准器"""
    
    def __init__(self):
        self.offset = 0  # 时间偏差（秒）
        
    def calibrate(self):
        """
        使用NTP协议校准时间
        返回：时间偏差（秒），正数表示本地时间快，负数表示慢
        """
        try:
            client = ntplib.NTPClient()
            # 尝试多个NTP服务器
            servers = ['pool.ntp.org', 'time.windows.com', 'cn.pool.ntp.org']
            
            for server in servers:
                try:
                    response = client.request(server, timeout=2)
                    self.offset = response.offset
                    return self.offset
                except:
                    continue
                    
            # 如果NTP都失败，使用HTTP方式
            return self._http_calibrate()
            
        except Exception as e:
            print(f"时间校准失败: {e}")
            return 0
    
    def _http_calibrate(self):
        """使用HTTP请求校准时间（备用方案）"""
        try:
            import requests
            response = requests.head('https://www.baidu.com', timeout=3)
            server_time_str = response.headers.get('Date')
            if server_time_str:
                from email.utils import parsedate_to_datetime
                server_time = parsedate_to_datetime(server_time_str)
                # 使用带时区的UTC时间进行比较
                local_time = datetime.now(timezone.utc)
                self.offset = (server_time - local_time).total_seconds()
                return self.offset
        except Exception as e:
            print(f"HTTP校准失败: {e}")
        return 0
    
    def get_adjusted_time(self):
        """获取校准后的当前时间"""
        return datetime.now() + timedelta(seconds=self.offset)
```

**精度目标**：偏差 < 0.1秒

**关键修复**：HTTP备用方案使用 `datetime.now(timezone.utc)` 替代 `datetime.utcnow()`，解决时区不一致导致的偏差计算错误

### 4.2 微信操作模块

```python
import pyautogui
import time
import pyperclip

class WeChatOperator:
    """微信操作类"""
    
    def __init__(self):
        pyautogui.FAILSAFE = True  # 启用故障安全（鼠标移到角落退出）
        pyautogui.PAUSE = 0.01     # 操作间隔10ms
    
    def find_and_open_chat(self, contact_name):
        """
        查找并打开聊天窗口
        步骤：
        1. 激活微信窗口
        2. Ctrl+F打开搜索
        3. 输入联系人名称
        4. 回车打开聊天
        """
        # 激活微信窗口（使用pywinauto或Alt+Tab）
        self._activate_wechat()
        
        # 打开搜索
        pyautogui.hotkey('ctrl', 'f')
        time.sleep(0.1)
        
        # 输入联系人名称
        pyautogui.typewrite(contact_name, interval=0.01)
        time.sleep(0.3)
        
        # 回车打开聊天
        pyautogui.press('enter')
        time.sleep(0.2)
        
        return True
    
    def prepare_message(self, message):
        """
        准备消息（输入但不发送）
        使用剪贴板支持中文，通过相对坐标点击输入框
        """
        import win32gui
        
        # 找到微信窗口
        hwnd = win32gui.FindWindow(None, "微信")
        if hwnd == 0:
            return False
        
        # 获取窗口位置，计算相对坐标
        left, top, right, bottom = win32gui.GetWindowRect(hwnd)
        width = right - left
        height = bottom - top
        
        # 计算输入框位置：窗口中央，底部往上100像素
        click_x = left + width // 2
        click_y = bottom - 100
        
        # 点击输入框
        pyautogui.click(click_x, click_y)
        time.sleep(0.1)
        
        # 清空输入框
        pyautogui.hotkey('ctrl', 'a')
        pyautogui.press('delete')
        time.sleep(0.1)
        
        # 使用剪贴板粘贴（支持中文）
        pyperclip.copy(message)
        pyautogui.hotkey('ctrl', 'v')
        time.sleep(0.2)
    
    def send_at_precise_time(self, target_time, calibrator):
        """
        在精确时间点发送消息，返回偏差（毫秒）
        """
        # 忙等待直到目标时间
        while True:
            now = calibrator.get_adjusted_time()
            
            if now >= target_time:
                break
            
            # 动态调整睡眠策略
            remaining = (target_time - now).total_seconds()
            
            if remaining > 1.0:
                time.sleep(0.1)
            elif remaining > 0.1:
                time.sleep(0.01)
            elif remaining > 0.005:
                time.sleep(0.001)
            # 小于5ms时忙等待
        
        # 执行发送
        pyautogui.press('enter')
        
        # 计算偏差（实际发送时间 - 目标时间）
        actual_time = calibrator.get_adjusted_time()
        deviation = (actual_time - target_time).total_seconds() * 1000  # 转为毫秒
        
        return True, deviation
        
    def _activate_wechat(self):
        """激活微信窗口到前台"""
        import pywinauto
        
        try:
            # 找到微信窗口
            app = pywinauto.Application().connect(title_re="微信")
            window = app.window(title_re="微信")
            window.set_focus()
            return True
        except:
            # 备用方案：使用Alt+Tab切换
            pyautogui.keyDown('alt')
            pyautogui.keyDown('tab')
            pyautogui.keyUp('tab')
            pyautogui.keyUp('alt')
            return False
```

### 4.3 任务调度器

```python
import threading
from datetime import datetime, timedelta

class TaskScheduler:
    """任务调度器（多线程管理）"""
    
    def __init__(self):
        self.tasks = []           # 任务列表
        self.threads = {}         # 运行中的线程
        self.time_offset = 0      # 时间偏差
        
    def add_task(self, contact, target_time, message):
        """添加定时任务"""
        task = {
            'id': len(self.tasks),
            'contact': contact,
            'target_time': target_time,
            'message': message,
            'status': 'pending'  # pending/running/completed/cancelled
        }
        self.tasks.append(task)
        return task['id']
    
    def start_task(self, task_id):
        """启动指定任务"""
        task = self.tasks[task_id]
        
        # 创建线程
        thread = threading.Thread(
            target=self._execute_task,
            args=(task,),
            name=f"Task-{task_id}"
        )
        thread.daemon = True
        thread.start()
        
        self.threads[task_id] = thread
        task['status'] = 'running'
        
    def _execute_task(self, task):
        """执行任务（在线程中运行）"""
        try:
            operator = WeChatOperator()
            
            # 计算准备时间（提前10秒）
            prepare_time = task['target_time'] - timedelta(seconds=10)
            
            # 等待到准备时间
            while datetime.now() < prepare_time:
                time.sleep(0.1)
                if task['status'] == 'cancelled':
                    return
            
            # 准备阶段：打开聊天窗口，输入消息
            self._log(f"[{task['contact']}] 准备发送...")
            operator.find_and_open_chat(task['contact'])
            operator.prepare_message(task['message'])
            
            # 精确等待到发送时间
            self._log(f"[{task['contact']}] 等待卡点...")
            operator.send_at_precise_time(
                task['target_time'], 
                offset=self.time_offset
            )
            
            task['status'] = 'completed'
            self._log(f"[{task['contact']}] 发送成功！")
            
        except Exception as e:
            task['status'] = 'error'
            self._log(f"[{task['contact']}] 发送失败: {str(e)}")
    
    def cancel_task(self, task_id):
        """取消任务"""
        if task_id in self.tasks:
            self.tasks[task_id]['status'] = 'cancelled'
            
    def _log(self, message):
        """记录日志（回调到GUI）"""
        timestamp = datetime.now().strftime("%H:%M:%S")
        print(f"[{timestamp}] {message}")
```

            self.task_list.takeItem(current_row)

        """添加日志"""
        timestamp = datetime.now().strftime("%H:%M:%S")
        self.log_area.append(f"[{timestamp}] {message}")

remaining = (target_time - now).total_seconds()

def safe_execute(func, max_retry=3):
    """带重试的执行"""
    for i in range(max_retry):
        try:
            return func()
        except Exception as e:
            if i == max_retry - 1:
                raise e
            time.sleep(0.5)

## 九、注意事项

## 十、已实现功能

### 10.1 窗口检测优化
```python
# 使用win32gui直接查找微信窗口
import win32gui

hwnd = win32gui.FindWindow(None, "微信")
if hwnd != 0:
    left, top, right, bottom = win32gui.GetWindowRect(hwnd)
```
**优化点**：直接通过窗口标题查找，避免多窗口匹配错误

### 10.2 相对坐标计算
```python
# 计算输入框位置（相对窗口坐标）
width = right - left
height = bottom - top
click_x = left + width // 2      # 窗口中央
click_y = bottom - 100           # 底部往上100像素
```
**优化点**：不依赖固定屏幕坐标，自适应窗口位置

### 10.3 发送偏差统计
```python
# 发送后计算实际偏差
actual_time = calibrator.get_adjusted_time()
deviation = (actual_time - target_time).total_seconds() * 1000
```
**功能**：发送完成后显示时间偏差（毫秒），便于评估精度

pyinstaller --onefile --windowed --name "微信定时消息发送器" wechat_timer.py
